# Script complet de pr√©paration dataset L√©ana Myst√®re
import os
import librosa
import soundfile as sf
import numpy as np
from pathlib import Path
import shutil

class DatasetPreparator:
    def __init__(self, input_folder, output_folder="dataset_leana", target_sr=40000):
        self.input_folder = Path(input_folder)
        self.output_folder = Path(output_folder)
        self.target_sr = target_sr
        
        # Cr√©ation structure dossiers
        self.output_folder.mkdir(exist_ok=True)
        
    def clean_audio(self, audio_path):
        """Nettoyage audio professionnel"""
        # Chargement audio
        y, sr = librosa.load(audio_path, sr=None)
        
        # Normalisation
        y = librosa.util.normalize(y)
        
        # Suppression silence d√©but/fin
        y, _ = librosa.effects.trim(y, top_db=20)
        
        # R√©duction bruit (simple)
        # Pour RVC, √©viter trop de traitement
        
        # R√©√©chantillonnage si n√©cessaire
        if sr != self.target_sr:
            y = librosa.resample(y, orig_sr=sr, target_sr=self.target_sr)
        
        return y, self.target_sr
    
    def segment_audio(self, y, sr, segment_duration=8.0):
        """Segmentation audio en chunks optimaux"""
        segment_samples = int(segment_duration * sr)
        segments = []
        
        # Division en segments
        for start in range(0, len(y), segment_samples):
            end = min(start + segment_samples, len(y))
            segment = y[start:end]
            
            # √âviter segments trop courts
            if len(segment) > segment_samples * 0.5:
                segments.append(segment)
        
        return segments
    
    def process_dataset(self):
        """Traitement complet du dataset"""
        audio_files = list(self.input_folder.glob("*.wav")) + \
                     list(self.input_folder.glob("*.mp3")) + \
                     list(self.input_folder.glob("*.m4a"))
        
        total_duration = 0
        file_count = 0
        
        print(f"üéµ Traitement de {len(audio_files)} fichiers audio...")
        
        for audio_file in audio_files:
            try:
                print(f"Traitement: {audio_file.name}")
                
                # Nettoyage audio
                y, sr = self.clean_audio(audio_file)
                
                # Segmentation
                segments = self.segment_audio(y, sr)
                
                # Sauvegarde segments
                for i, segment in enumerate(segments):
                    output_path = self.output_folder / f"leana_{file_count:03d}_{i:02d}.wav"
                    sf.write(output_path, segment, sr)
                    
                    duration = len(segment) / sr
                    total_duration += duration
                    
                    print(f"  ‚úÖ Segment {i+1}: {duration:.1f}s")
                
                file_count += 1
                
            except Exception as e:
                print(f"‚ùå Erreur avec {audio_file.name}: {e}")
        
        print(f"\nüéâ Dataset pr√©par√©!")
        print(f"üìÅ Dossier: {self.output_folder}")
        print(f"üìä {len(list(self.output_folder.glob('*.wav')))} fichiers")
        print(f"‚è±Ô∏è Dur√©e totale: {total_duration/60:.1f} minutes")
        
        return self.output_folder

# Utilisation

import librosa
import numpy as np
import matplotlib.pyplot as plt

def analyze_conversion_quality(original, converted):
    """Analyse compar√©e audio original vs converti"""
    
    # Chargement audios
    y1, sr1 = librosa.load(original)
    y2, sr2 = librosa.load(converted)
    
    # Spectrogrammes
    S1 = librosa.stft(y1)
    S2 = librosa.stft(y2)
    
    # M√©triques qualit√©
    spectral_similarity = np.corrcoef(
        np.abs(S1).flatten(), 
        np.abs(S2).flatten()
    )[0,1]
    
    # Analyse pitch
    pitches1, magnitudes1 = librosa.piptrack(y=y1, sr=sr1)
    pitches2, magnitudes2 = librosa.piptrack(y=y2, sr=sr2)
    
    print(f"üìä Similarit√© spectrale: {spectral_similarity:.3f}")
    print(f"üéµ Coh√©rence pitch: {np.mean(pitches2/pitches1):.3f}")
    
    return spectral_similarity

# Utilisation

# Post-traitement audio RVC
import soundfile as sf
from scipy import signal

def enhance_rvc_output(input_path, output_path):
    """Am√©lioration audio post-RVC"""
    
    # Chargement
    y, sr = librosa.load(input_path, sr=None)
    
    # 1. Normalisation douce
    y = librosa.util.normalize(y, norm=np.inf)
    
    # 2. √âgalisation subtile
    # Boost l√©ger dans m√©diums (voix)
    sos = signal.butter(2, [200, 3000], 
                       btype='band', fs=sr, 
                       output='sos')
    y_eq = signal.sosfilt(sos, y) * 0.1
    y = y + y_eq
    
    # 3. Compression douce
    threshold = 0.7
    ratio = 3.0
    y = np.where(np.abs(y) > threshold,
                np.sign(y) * (threshold + 
                (np.abs(y) - threshold) / ratio),
                y)
    
    # 4. D√©-essage (r√©duction sifflantes)
    sos_deess = signal.butter(2, [5000, 8000], 
                             btype='band', fs=sr, 
                             output='sos')
    sibilants = signal.sosfilt(sos_deess, y)
    y = y - sibilants * 0.3
    
    # Sauvegarde
    sf.write(output_path, y, sr)
    print(f"‚úÖ Post-traitement termin√©: {output_path}")

# Application
enhance_rvc_output("raw_conversion.wav", "enhanced_leana.wav")

# Syst√®me batch processing avanc√© pour RVC
import concurrent.futures
import threading
from queue import Queue
import time

class RVCBatchProcessor:
    def __init__(self, model_path, index_path, max_workers=2):
        self.model_path = model_path
        self.index_path = index_path
        self.max_workers = max_workers
        self.converter = RVCConverter(model_path, index_path)
        self.results_queue = Queue()
        
    def process_file(self, file_info):
        """Traitement d'un fichier individuel"""
        input_file, output_file, params = file_info
        
        start_time = time.time()
        
        try:
            success = self.converter.convert_voice(
                input_file, 
                output_file,
                pitch_change=params.get('pitch', 0),
                index_rate=params.get('index_rate', 0.7)
            )
            
            conversion_time = time.time() - start_time
            
            result = {
                'input': input_file,
                'output': output_file,
                'success': success,
                'time': conversion_time,
                'thread': threading.current_thread().name
            }
            
            self.results_queue.put(result)
            return result
            
        except Exception as e:
            error_result = {
                'input': input_file,
                'error': str(e),
                'success': False,
                'time': time.time() - start_time
            }
            self.results_queue.put(error_result)
            return error_result
    
    def process_batch(self, file_list, params_list=None):
        """Traitement batch multi-thread"""
        if params_list is None:
            params_list = [{}] * len(file_list)
        
        # Pr√©paration t√¢ches
        tasks = [(f, f.replace('.wav', '_leana.wav'), p) 
                for f, p in zip(file_list, params_list)]
        
        print(f"üöÄ D√©marrage batch: {len(tasks)} fichiers")
        print(f"üîß Workers: {self.max_workers}")
        
        start_time = time.time()
        
        # Ex√©cution parall√®le
        with concurrent.futures.ThreadPoolExecutor(max_workers=self.max_workers) as executor:
            futures = [executor.submit(self.process_file, task) for task in tasks]
            
            # Collecte r√©sultats
            results = []
            for future in concurrent.futures.as_completed(futures):
                result = future.result()
                results.append(result)
                
                if result['success']:
                    print(f"‚úÖ {result['input']} ‚Üí {result['time']:.1f}s")
                else:
                    print(f"‚ùå {result['input']} ‚Üí ERREUR")
        
        total_time = time.time() - start_time
        successful = sum(1 for r in results if r['success'])
        
        print(f"\nüìä BATCH TERMIN√â:")
        print(f"‚úÖ R√©ussis: {successful}/{len(tasks)}")
        print(f"‚è±Ô∏è  Temps total: {total_time:.1f}s")
        print(f"‚ö° Temps moyen: {total_time/len(tasks):.1f}s/fichier")
        
        return results
