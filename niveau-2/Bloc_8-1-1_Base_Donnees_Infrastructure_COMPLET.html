import sqlite3
import os

class MYMDatabaseManager:
    def __init__(self, db_path="mym_automation.db"):
        self.db_path = db_path
        self.connection = None
        self.initialize_database()
    
    def initialize_database(self):
        """Initialisation complète de la base de données"""
        self.connection = sqlite3.connect(self.db_path)
        self.connection.row_factory = sqlite3.Row
        
        self.create_content_library_table()
        self.create_publication_schedule_table()
        self.create_subscriber_interactions_table()
        self.create_analytics_data_table()
        self.create_system_config_table()
        self.create_indexes()
        
        print("✅ Base de données MYM initialisée avec succès")

Table: content_library 

def create_content_library_table(self):
    """Création table bibliothèque de contenu"""
    create_sql = """
    CREATE TABLE IF NOT EXISTS content_library (
        content_id TEXT PRIMARY KEY,
        content_type TEXT NOT NULL CHECK(content_type IN ('image', 'video', 'text', 'story')),
        file_path TEXT NOT NULL,
        title TEXT NOT NULL,
        description TEXT,
        tags TEXT, -- JSON array of tags
        nsfw_level INTEGER CHECK(nsfw_level BETWEEN 1 AND 10),
        persona_id TEXT DEFAULT 'leana_mystere',
        
        -- Métadonnées techniques
        file_size_mb REAL,
        resolution TEXT, -- "1024x1024"
        duration_seconds INTEGER, -- pour vidéos
        
        -- Métadonnées de génération
        prompt_used TEXT,
        negative_prompt TEXT,
        model_used TEXT,
        generation_settings TEXT, -- JSON
        
        -- Status et validation
        status TEXT DEFAULT 'generated' CHECK(status IN ('generated', 'validated', 'published', 'archived')),
        quality_score REAL DEFAULT 0.0,
        validation_notes TEXT,
        
        -- Données de performance
        views_count INTEGER DEFAULT 0,
        likes_count INTEGER DEFAULT 0,
        comments_count INTEGER DEFAULT 0,
        revenue_generated REAL DEFAULT 0.0,
        
        -- Timestamps
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
        published_at TIMESTAMP,
        last_modified TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
        
        -- Backup et sécurité
        backup_location TEXT,
        encryption_status TEXT DEFAULT 'encrypted',
        access_level TEXT DEFAULT 'premium' CHECK(access_level IN ('free', 'premium', 'vip', 'custom'))
    );
    """
    
    self.connection.execute(create_sql)
    self.connection.commit()
    print("✅ Table content_library créée")

Table: publication_schedule 

def create_publication_schedule_table(self):
    """Création table planning de publication"""
    create_sql = """
    CREATE TABLE IF NOT EXISTS publication_schedule (
        schedule_id TEXT PRIMARY KEY,
        content_id TEXT NOT NULL,
        
        -- Planning
        scheduled_datetime TIMESTAMP NOT NULL,
        platform TEXT DEFAULT 'mym' CHECK(platform IN ('mym', 'onlyfans', 'instagram', 'twitter')),
        publication_type TEXT CHECK(publication_type IN ('post', 'story', 'live', 'message')),
        
        -- Configuration publication
        caption TEXT,
        hashtags TEXT, -- JSON array
        pricing_tier TEXT CHECK(pricing_tier IN ('free', 'premium', 'vip', 'ppv')),
        custom_price REAL,
        
        -- Audience et ciblage
        target_audience TEXT, -- JSON: {"age_min": 18, "age_max": 45, "gender": "all"}
        geographic_targeting TEXT, -- JSON array de pays/régions
        
        -- Status et tracking
        status TEXT DEFAULT 'scheduled' CHECK(status IN ('scheduled', 'publishing', 'published', 'failed', 'cancelled')),
        attempts_count INTEGER DEFAULT 0,
        last_attempt_at TIMESTAMP,
        error_message TEXT,
        
        -- Analytics et performance
        engagement_predicted REAL, -- Score prédictif 0-1
        optimal_time_score REAL, -- Score de l'heure optimale
        competition_level TEXT CHECK(competition_level IN ('low', 'medium', 'high')),
        
        -- Automation
        auto_publish BOOLEAN DEFAULT 1,
        retry_on_failure BOOLEAN DEFAULT 1,
        max_retry_attempts INTEGER DEFAULT 3,
        
        -- Timestamps
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
        published_at TIMESTAMP,
        
        -- Relations
        FOREIGN KEY (content_id) REFERENCES content_library (content_id)
    );
    """
    
    self.connection.execute(create_sql)
    self.connection.commit()
    print("✅ Table publication_schedule créée")

Table: subscriber_interactions 

def create_subscriber_interactions_table(self):
    """Création table interactions abonnés"""
    create_sql = """
    CREATE TABLE IF NOT EXISTS subscriber_interactions (
        interaction_id TEXT PRIMARY KEY,
        subscriber_id TEXT NOT NULL,
        
        -- Informations abonné
        subscriber_username TEXT,
        subscriber_tier TEXT CHECK(subscriber_tier IN ('free', 'premium', 'vip', 'whale')),
        subscription_date TIMESTAMP,
        total_spent REAL DEFAULT 0.0,
        
        -- Données interaction
        interaction_type TEXT CHECK(interaction_type IN ('message', 'like', 'comment', 'tip', 'purchase', 'view')),
        content_id TEXT, -- Peut être NULL pour interactions générales
        message_content TEXT,
        
        -- Analyse sentiment et intention
        sentiment_score REAL, -- -1 à 1 (négatif à positif)
        intent_category TEXT, -- 'compliment', 'request', 'complaint', 'flirt', 'business'
        keywords_detected TEXT, -- JSON array
        
        -- Réponse automatique
        auto_response_sent BOOLEAN DEFAULT 0,
        response_template_used TEXT,
        response_content TEXT,
        response_timestamp TIMESTAMP,
        
        -- Données comportementales
        session_duration INTEGER, -- en secondes
        pages_viewed INTEGER,
        last_seen TIMESTAMP,
        device_type TEXT, -- 'mobile', 'desktop', 'tablet'
        
        -- Scoring et segmentation
        engagement_score REAL DEFAULT 0.0, -- 0-100
        loyalty_score REAL DEFAULT 0.0, -- 0-100  
        monetization_potential REAL DEFAULT 0.0, -- 0-100
        risk_score REAL DEFAULT 0.0, -- 0-100 (spam, problématique)
        
        -- Préférences détectées
        content_preferences TEXT, -- JSON: {"nsfw_level": [5,7], "types": ["image", "video"]}
        interaction_frequency TEXT, -- 'daily', 'weekly', 'monthly', 'occasional'
        peak_activity_hours TEXT, -- JSON array [18, 19, 20, 21]
        
        -- Timestamps
        interaction_timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
        
        -- Relations
        FOREIGN KEY (content_id) REFERENCES content_library (content_id)
    );
    """
    
    self.connection.execute(create_sql)
    self.connection.commit()
    print("✅ Table subscriber_interactions créée")

Étape 1: Vérification des Prérequis

Prérequis Obligatoires

  • Python 3.8+ installé
  • Modules sqlite3 et json (inclus)
  • Espace disque minimum: 5GB
  • Permissions d'écriture sur le disque
# Vérification prérequis système
import sys
import sqlite3
import os
import shutil

def check_prerequisites():
    """Vérification complète des prérequis"""
    checks = []
    
    # Version Python
    if sys.version_info >= (3, 8):
        checks.append(("Python 3.8+", True))
    else:
        checks.append(("Python 3.8+", False))
    
    # SQLite disponible
    try:
        sqlite3.connect(":memory:").close()
        checks.append(("SQLite3", True))
    except:
        checks.append(("SQLite3", False))
    
    # Espace disque
    free_space_gb = shutil.disk_usage(".")[2] / (1024**3)
    checks.append((f"Espace disque ({free_space_gb:.1f}GB)", free_space_gb >= 5))
    
    # Affichage résultats
    print("🔍 Vérification des prérequis:")
    for check_name, passed in checks:
        status = "✅" if passed else "❌"
        print(f"  {status} {check_name}")
    
    return all(check[1] for check in checks)

# Exécution
if check_prerequisites():
    print("✅ Tous les prérequis sont satisfaits!")
else:
    print("❌ Certains prérequis ne sont pas satisfaits")

import os

import json

def install_mym_database():
    """Installation complète de la base MYM"""
    print("🚀 Début installation base de données MYM...")
    
    # 1. Création du dossier de données
    data_dir = "mym_data"
    if not os.path.exists(data_dir):
        os.makedirs(data_dir)
        print(f"📁 Dossier {data_dir} créé")
    
    # 2. Sous-dossiers organisationnels
    subdirs = ["content", "backups", "exports", "logs"]
    for subdir in subdirs:
        full_path = os.path.join(data_dir, subdir)
        if not os.path.exists(full_path):
            os.makedirs(full_path)
            print(f"📁 Sous-dossier {subdir} créé")
    
    # 3. Initialisation de la base
    db_path = os.path.join(data_dir, "mym_automation.db")
    db_manager = MYMDatabaseManager(db_path)
    
    # 4. Insertion de données de test
    db_manager.insert_sample_data()
    
    # 5. Vérification de l'installation
    if db_manager.verify_installation():
        print("✅ Installation terminée avec succès!")
        print(f"📊 Base de données: {db_path}")
        return True
    else:
        print("❌ Erreur lors de l'installation")
        return False

# Lancement installation
if __name__ == "__main__":
    install_mym_database()

Classe MYMDatabaseManager Complète 

class MYMDatabaseManager:
    """Manager principal pour toutes les opérations de base de données"""
    
    def __init__(self, db_path="mym_automation.db"):
        self.db_path = db_path
        self.connection = None
        self.initialize_database()
    
    def add_content(self, content_data):
        """Ajout nouveau contenu à la bibliothèque"""
        sql = """
        INSERT INTO content_library (
            content_id, content_type, file_path, title, description,
            tags, nsfw_level, persona_id, prompt_used, model_used,
            quality_score, status
        ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
        """
        
        try:
            self.connection.execute(sql, (
                content_data['content_id'],
                content_data['content_type'],
                content_data['file_path'],
                content_data['title'],
                content_data.get('description', ''),
                json.dumps(content_data.get('tags', [])),
                content_data.get('nsfw_level', 1),
                content_data.get('persona_id', 'leana_mystere'),
                content_data.get('prompt_used', ''),
                content_data.get('model_used', ''),
                content_data.get('quality_score', 0.0),
                content_data.get('status', 'generated')
            ))
            self.connection.commit()
            print(f"✅ Contenu ajouté: {content_data['content_id']}")
            return True
        except sqlite3.Error as e:
            print(f"❌ Erreur ajout contenu: {e}")
            return False
    
    def schedule_publication(self, schedule_data):
        """Planification d'une publication"""
        sql = """
        INSERT INTO publication_schedule (
            schedule_id, content_id, scheduled_datetime, platform,
            publication_type, caption, pricing_tier, status
        ) VALUES (?, ?, ?, ?, ?, ?, ?, ?)
        """
        
        try:
            self.connection.execute(sql, (
                schedule_data['schedule_id'],
                schedule_data['content_id'],
                schedule_data['scheduled_datetime'],
                schedule_data.get('platform', 'mym'),
                schedule_data.get('publication_type', 'post'),
                schedule_data.get('caption', ''),
                schedule_data.get('pricing_tier', 'premium'),
                'scheduled'
            ))
            self.connection.commit()
            print(f"✅ Publication planifiée: {schedule_data['schedule_id']}")
            return True
        except sqlite3.Error as e:
            print(f"❌ Erreur planification: {e}")
            return False
    
    def get_pending_publications(self):
        """Récupération des publications en attente"""
        sql = """
        SELECT ps.*, cl.file_path, cl.title, cl.content_type
        FROM publication_schedule ps
        JOIN content_library cl ON ps.content_id = cl.content_id
        WHERE ps.status = 'scheduled' 
        AND ps.scheduled_datetime <= datetime('now', '+1 hour')
        ORDER BY ps.scheduled_datetime ASC
        """
        
        cursor = self.connection.execute(sql)
        results = cursor.fetchall()
        
        return [dict(row) for row in results]
    
    def log_interaction(self, interaction_data):
        """Enregistrement d'une interaction abonné"""
        sql = """
        INSERT INTO subscriber_interactions (
            interaction_id, subscriber_id, subscriber_username,
            interaction_type, message_content, sentiment_score,
            engagement_score
        ) VALUES (?, ?, ?, ?, ?, ?, ?)
        """
        
        try:
            self.connection.execute(sql, (
                interaction_data['interaction_id'],
                interaction_data['subscriber_id'],
                interaction_data.get('subscriber_username', 'Anonymous'),
                interaction_data['interaction_type'],
                interaction_data.get('message_content', ''),
                interaction_data.get('sentiment_score', 0.0),
                interaction_data.get('engagement_score', 0.0)
            ))
            self.connection.commit()
            return True
        except sqlite3.Error as e:
            print(f"❌ Erreur log interaction: {e}")
            return False

Backup Automatique 

def create_backup(self):
    """Création backup automatique"""
    import shutil
    from datetime import datetime
    
    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    backup_name = f"mym_backup_{timestamp}.db"
    backup_path = os.path.join("mym_data/backups", backup_name)
    
    try:
        shutil.copy2(self.db_path, backup_path)
        print(f"✅ Backup créé: {backup_path}")
        return backup_path
    except Exception as e:
        print(f"❌ Erreur backup: {e}")
        return None

Statistiques Base 

def get_database_stats(self):
    """Statistiques de la base"""
    stats = {}
    
    # Contenu total
    cursor = self.connection.execute(
        "SELECT COUNT(*) FROM content_library"
    )
    stats['total_content'] = cursor.fetchone()[0]
    
    # Publications programmées
    cursor = self.connection.execute(
        "SELECT COUNT(*) FROM publication_schedule WHERE status='scheduled'"
    )
    stats['scheduled_posts'] = cursor.fetchone()[0]
    
    # Interactions aujourd'hui
    cursor = self.connection.execute(
        "SELECT COUNT(*) FROM subscriber_interactions WHERE DATE(interaction_timestamp) = DATE('now')"
    )
    stats['today_interactions'] = cursor.fetchone()[0]
    
    return stats
