# Script Python - Préparation Audio pour ElevenLabs
import librosa
import soundfile as sf
import numpy as np
from scipy.signal import butter, filtfilt

def prepare_audio_for_cloning(input_file, output_file):
    """Prépare un fichier audio pour le clonage vocal"""
    
    # Chargement audio
    audio, sr = librosa.load(input_file, sr=22050)
    
    # Normalisation du volume
    audio = librosa.util.normalize(audio)
    
    # Suppression du silence
    audio_trimmed, _ = librosa.effects.trim(audio, top_db=20)
    
    # Filtre passe-bas pour réduire le bruit haute fréquence
    nyquist = sr / 2
    cutoff = 8000  # Fréquence de coupure
    b, a = butter(5, cutoff / nyquist, btype='low')
    audio_filtered = filtfilt(b, a, audio_trimmed)
    
    # Sauvegarde
    sf.write(output_file, audio_filtered, sr)
    
    # Informations
    duration = len(audio_filtered) / sr
    print(f"✅ Audio préparé: {duration:.1f}s, {sr}Hz")
    
    return output_file

# Utilisation
input_file = "voix_originale.wav"
output_file = "voix_preparee_elevenlabs.wav"
prepare_audio_for_cloning(input_file, output_file)

# Script Python - Enregistrement Audio Optimisé
import pyaudio
import wave
import numpy as np
import time

class AudioRecorder:
    def __init__(self, sample_rate=44100, chunk_size=1024):
        self.sample_rate = sample_rate
        self.chunk_size = chunk_size
        self.format = pyaudio.paInt16
        self.channels = 1  # Mono
        
    def record_session(self, filename, duration_seconds):
        """Enregistre une session audio optimisée pour clonage vocal"""
        
        audio = pyaudio.PyAudio()
        
        # Configuration stream
        stream = audio.open(
            format=self.format,
            channels=self.channels,
            rate=self.sample_rate,
            input=True,
            frames_per_buffer=self.chunk_size
        )
        
        print("🎤 Enregistrement démarré...")
        print(f"⏱️ Durée: {duration_seconds} secondes")
        
        frames = []
        for i in range(0, int(self.sample_rate / self.chunk_size * duration_seconds)):
            data = stream.read(self.chunk_size)
            frames.append(data)
            
            # Monitoring niveau
            audio_data = np.frombuffer(data, dtype=np.int16)
            level = np.abs(audio_data).mean()
            
            if i % 10 == 0:  # Affichage toutes les 0.2s
                level_db = 20 * np.log10(level / 32767.0) if level > 0 else -100
                print(f"📊 Niveau: {level_db:.1f}dB", end='\r')
        
        print("\n✅ Enregistrement terminé")
        
        # Sauvegarde
        stream.stop_stream()
        stream.close()
        audio.terminate()
        
        # Création fichier WAV
        with wave.open(filename, 'wb') as wf:
            wf.setnchannels(self.channels)
            wf.setsampwidth(audio.get_sample_size(self.format))
            wf.setframerate(self.sample_rate)
            wf.writeframes(b''.join(frames))
        
        return filename

# Utilisation
recorder = AudioRecorder()
audio_file = recorder.record_session("voix_source.wav", 300)  # 5 minutes

# Script Python - Post-Traitement Audio Complet
import librosa
import soundfile as sf
import numpy as np
from scipy.signal import butter, filtfilt
import noisereduce as nr

class AudioProcessor:
    def __init__(self):
        self.target_sr = 22050  # Optimal pour la plupart des modèles
        
    def process_for_cloning(self, input_file, output_file):
        """Pipeline complet de traitement audio"""
        
        print(f"🎵 Traitement de {input_file}...")
        
        # 1. Chargement
        audio, sr = librosa.load(input_file, sr=None)
        print(f"📊 Audio original: {len(audio)/sr:.1f}s, {sr}Hz")
        
        # 2. Resampling si nécessaire
        if sr != self.target_sr:
            audio = librosa.resample(audio, orig_sr=sr, target_sr=self.target_sr)
            sr = self.target_sr
            print(f"🔄 Resampling vers {sr}Hz")
        
        # 3. Suppression bruit de fond
        audio_denoised = nr.reduce_noise(y=audio, sr=sr)
        print("🧹 Suppression bruit de fond")
        
        # 4. Suppression silences
        audio_trimmed, _ = librosa.effects.trim(
            audio_denoised, 
            top_db=20,
            frame_length=2048,
            hop_length=512
        )
        print("✂️ Suppression silences")
        
        # 5. Normalisation RMS
        target_rms = 0.2  # -14dB RMS approximativement
        current_rms = np.sqrt(np.mean(audio_trimmed**2))
        if current_rms > 0:
            audio_normalized = audio_trimmed * (target_rms / current_rms)
        else:
            audio_normalized = audio_trimmed
        print("📈 Normalisation RMS")
        
        # 6. Limitation douce (soft limiting)
        audio_limited = np.tanh(audio_normalized * 0.9) * 0.95
        print("🎚️ Limitation douce")
        
        # 7. Filtre passe-bas léger (anti-aliasing)
        nyquist = sr / 2
        cutoff = 8000  # Coupe à 8kHz
        b, a = butter(5, cutoff / nyquist, btype='low')
        audio_filtered = filtfilt(b, a, audio_limited)
        print("🔽 Filtrage passe-bas")
        
        # 8. Vérification finale
        peak_level = np.max(np.abs(audio_filtered))
        rms_level = np.sqrt(np.mean(audio_filtered**2))
        duration = len(audio_filtered) / sr
        
        print(f"✅ Traitement terminé:")
        print(f"   Durée: {duration:.1f}s")
        print(f"   Peak: {20*np.log10(peak_level):.1f}dB")
        print(f"   RMS: {20*np.log10(rms_level):.1f}dB")
        
        # 9. Sauvegarde
        sf.write(output_file, audio_filtered, sr)
        print(f"💾 Sauvegardé: {output_file}")
        
        return {
            'duration': duration,
            'peak_db': 20*np.log10(peak_level),
            'rms_db': 20*np.log10(rms_level),
            'sample_rate': sr
        }
    
    def validate_quality(self, audio_file):
        """Valide la qualité audio pour clonage"""
        
        audio, sr = librosa.load(audio_file)
        duration = len(audio) / sr
        
        # Critères de validation
        checks = {
            'duration_ok': 60 <= duration <= 600,  # 1-10 minutes
            'no_clipping': np.max(np.abs(audio)) < 0.99,
            'sufficient_energy': np.sqrt(np.mean(audio**2)) > 0.01,
            'proper_sr': sr in [22050, 44100, 48000]
        }
        
        score = sum(checks.values()) / len(checks)
        
        print(f"🎯 Score qualité: {score*100:.0f}%")
        for check, passed in checks.items():
            status = "✅" if passed else "❌"
            print(f"   {status} {check}: {passed}")
        
        return score >= 0.75

# Script Python - Détection Voix Clonée
import librosa
import numpy as np
from scipy import signal
from scipy.stats import entropy
import requests

class VoiceAuthenticityDetector:
    def __init__(self):
        self.features_weights = {
            'spectral_irregularity': 0.25,
            'breathing_detection': 0.20,
            'formant_consistency': 0.20,
            'temporal_patterns': 0.15,
            'compression_artifacts': 0.10,
            'pitch_naturalness': 0.10
        }
    
    def analyze_audio(self, audio_file):
        """Analyse complète d'authenticité audio"""
        
        audio, sr = librosa.load(audio_file, sr=22050)
        
        results = {
            'spectral_irregularity': self.check_spectral_irregularity(audio, sr),
            'breathing_detection': self.detect_breathing_patterns(audio, sr),
            'formant_consistency': self.analyze_formant_consistency(audio, sr),
            'temporal_patterns': self.check_temporal_patterns(audio, sr),
            'compression_artifacts': self.detect_compression_artifacts(audio, sr),
            'pitch_naturalness': self.analyze_pitch_naturalness(audio, sr)
        }
        
        # Score composite
        authenticity_score = sum(
            results[feature] * weight 
            for feature, weight in self.features_weights.items()
        )
        
        confidence_level = self.calculate_confidence(results, authenticity_score)
        
        return {
            'authenticity_score': authenticity_score,
            'confidence_level': confidence_level,
            'detailed_analysis': results,
            'verdict': self.get_verdict(authenticity_score, confidence_level)
        }
    
    def check_spectral_irregularity(self, audio, sr):
        """Vérifie les irrégularités spectrales typiques de l'IA"""
        
        # Spectrogramme
        D = librosa.stft(audio, n_fft=2048, hop_length=512)
        magnitude = np.abs(D)
        
        # Analyse bandes de fréquence
        freq_bins = magnitude.shape[0]
        band_energies = []
        
        for i in range(0, freq_bins, freq_bins//10):
            band_energy = np.mean(magnitude[i:i+freq_bins//10])
            band_energies.append(band_energy)
        
        # Irrégularités entre bandes (IA tend à créer des patterns non naturels)
        irregularity = np.std(band_energies) / np.mean(band_energies)
        
        # Score : plus l'irrégularité est faible, plus c'est suspect
        return min(irregularity * 2, 1.0)
    
    def detect_breathing_patterns(self, audio, sr):
        """Détecte la présence de respirations naturelles"""
        
        # Filtrage basses fréquences pour isoler respirations
        b, a = signal.butter(4, 300 / (sr/2), btype='low')
        low_freq = signal.filtfilt(b, a, audio)
        
        # Détection de patterns périodiques de faible amplitude
        # (respirations entre phrases)
        envelope = np.abs(signal.hilbert(low_freq))
        smooth_envelope = signal.savgol_filter(envelope, 441, 3)
        
        # Recherche de variations périodiques typiques de respiration
        breathing_peaks, _ = signal.find_peaks(
            smooth_envelope, 
            height=np.max(smooth_envelope) * 0.1,
            distance=sr//4  # Minimum 0.25s entre respirations
        )
        
        # Score basé sur la fréquence de respiration (humain: 12-20/min)
        duration_min = len(audio) / sr / 60
        breathing_rate = len(breathing_peaks) / duration_min
        
        if 8 <= breathing_rate <= 25:  # Plage naturelle élargie
            return 1.0
        elif 5 <= breathing_rate <= 30:
            return 0.7
        else:
            return 0.2  # Suspect
    
    def analyze_formant_consistency(self, audio, sr):
        """Analyse la cohérence des formants vocaux"""
        
        # Extraction des formants
        frame_length = int(0.025 * sr)  # 25ms
        hop_length = int(0.01 * sr)     # 10ms
        
        formants = []
        for i in range(0, len(audio) - frame_length, hop_length):
            frame = audio[i:i + frame_length]
            
            if np.std(frame) > 0.01:  # Frame avec contenu vocal
                # Estimation formants via LPC
                try:
                    lpc_coeffs = librosa.lpc(frame, order=12)
                    roots = np.roots(lpc_coeffs)
                    
                    # Conversion en fréquences
                    freqs = []
                    for root in roots:
                        if np.imag(root) > 0:
                            freq = np.angle(root) * sr / (2 * np.pi)
                            if 100 <= freq <= 4000:  # Plage formants vocaux
                                freqs.append(freq)
                    
                    if len(freqs) >= 2:
                        formants.append(sorted(freqs)[:3])  # 3 premiers formants
                        
                except:
                    continue
        
        if len(formants) < 10:
            return 0.5  # Pas assez de données
        
        # Analyse de la variabilité des formants
        formants_array = np.array(formants)
        if formants_array.size == 0:
            return 0.5
        
        # Coefficient de variation pour chaque formant
        cv_formants = []
        for f in range(min(3, formants_array.shape[1])):
            if formants_array.shape[1] > f:
                std_f = np.std(formants_array[:, f])
                mean_f = np.mean(formants_array[:, f])
                if mean_f > 0:
                    cv_formants.append(std_f / mean_f)
        
        # Score basé sur variabilité naturelle
        avg_cv = np.mean(cv_formants) if cv_formants else 0
        
        # Variabilité naturelle attendue: 0.1-0.3
        if 0.05 <= avg_cv <= 0.4:
            return 1.0
        elif avg_cv < 0.02:  # Trop constant = suspect
            return 0.3
        else:  # Trop variable = possiblement naturel mais degradé
            return 0.7
    
    def get_verdict(self, score, confidence):
        """Détermine le verdict final"""
        
        if confidence < 0.6:
            return {
                'status': 'UNCERTAIN',
                'message': 'Analyse non concluante - plus de données nécessaires',
                'color': 'yellow'
            }
        elif score >= 0.7:
            return {
                'status': 'AUTHENTIC',
                'message': 'Voix probablement authentique',
                'color': 'green'
            }
        elif score >= 0.4:
            return {
                'status': 'SUSPICIOUS',
                'message': 'Voix suspecte - vérification recommandée',
                'color': 'orange'
            }
        else:
            return {
                'status': 'SYNTHETIC',
                'message': 'Voix probablement générée par IA',
                'color': 'red'
            }
