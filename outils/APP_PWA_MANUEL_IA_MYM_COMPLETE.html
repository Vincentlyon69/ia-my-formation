class DataProtectionSystem: def __init__(self): self.encryption_key = self.generate_master_key() self.backup_locations = ['local_encrypted', 'cloud_encrypted', 'offline_backup'] def create_encrypted_vault(self, vault_path, size_gb, password): # Création conteneur chiffré VeraCrypt veracrypt_cmd = f""" veracrypt -t -c --volume-type=normal --filesystem=NTFS --encryption=AES --hash=SHA-512 --size={size_gb}GB --password="{password}" --pim=0 --keyfiles="" --random-source=/dev/urandom "{vault_path}" """ subprocess.run(veracrypt_cmd, shell=True) return vault_path def secure_file_deletion(self, file_path): # Suppression sécurisée avec overwrite import os import random if os.path.exists(file_path): filesize = os.path.getsize(file_path) # 3 passes d'écrasement with open(file_path, "r+b") as f: for _ in range(3): f.seek(0) f.write(os.urandom(filesize)) f.flush() os.fsync(f.fileno()) os.remove(file_path) return True
