        """

    def calculate_persona_compatibility(self, persona_a: str, persona_b: str) -> Dict:
        """
        üßÆ CALCUL DE COMPATIBILIT√â ENTRE PERSONAS
        
        Analyse la compatibilit√© entre deux personas selon plusieurs crit√®res.

        config_a = self.manager.personas[persona_a]
        config_b = self.manager.personas[persona_b]
        
        # 1. Overlap d'audience (plus c'est √©lev√©, plus il y a risque de conflit)
        audience_overlap = self.calculate_audience_overlap(config_a, config_b)
        
        # 2. Conflit de niche (m√™me sp√©cialisation = conflit)

        # 3. Potentiel de synergie (compl√©mentarit√©)
        synergy_score = self.calculate_synergy_potential(config_a, config_b)
        
        return {
            'audience_overlap': audience_overlap,
            'conflict_risk': (audience_overlap + niche_conflict) / 2,
            'synergy_potential': synergy_score,
            'recommended_action': self.get_compatibility_recommendation(
                audience_overlap, niche_conflict, synergy_score

    def calculate_audience_overlap(self, config_a, config_b) -> float:
        """Calcule le pourcentage d'audience partag√©e entre deux personas"""
        # Simulation bas√©e sur les cat√©gories et march√©s cibles
        same_market = config_a.target_market == config_b.target_market
        same_age_range = abs(config_a.age - config_b.age) < 5
        same_language = config_a.primary_language == config_b.primary_language
        
        overlap_score = 0
        if same_market: overlap_score += 0.4
        if same_age_range: overlap_score += 0.3
        if same_language: overlap_score += 0.3
        
        return min(overlap_score, 1.0)
        
    def calculate_niche_conflict(self, config_a, config_b) -> float:
        """Calcule le risque de conflit de niche"""
        # Conflits bas√©s sur les sp√©cialisations
        specializations_a = set(config_a.specialization)
        specializations_b = set(config_b.specialization)
        
        common_specs = specializations_a.intersection(specializations_b)
        total_specs = specializations_a.union(specializations_b)
        
        if len(total_specs) == 0:
            return 0.0
            
        conflict_ratio = len(common_specs) / len(total_specs)
        return conflict_ratio
        
    def calculate_synergy_potential(self, config_a, config_b) -> float:
        """Calcule le potentiel de synergie entre deux personas"""
        # Synergies bas√©es sur la compl√©mentarit√©
        complementary_categories = {
            'artistic': ['fitness', 'luxury'],
            'fitness': ['artistic', 'mainstream'],
            'mature': ['luxury', 'artistic'],
            'luxury': ['artistic', 'mature']
        }
        
        cat_a = config_a.category.value if hasattr(config_a.category, 'value') else str(config_a.category)
        cat_b = config_b.category.value if hasattr(config_b.category, 'value') else str(config_b.category)
        
        if cat_b in complementary_categories.get(cat_a, []):
            return 0.8  # Forte synergie
        elif cat_a != cat_b:
            return 0.5  # Synergie mod√©r√©e
        else:
            return 0.2  # Faible synergie (m√™me cat√©gorie)
