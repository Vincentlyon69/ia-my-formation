def create_publication_schedule_table(self):

    CREATE TABLE IF NOT EXISTS publication_schedule (

        attempts_count INTEGER DEFAULT 0,

        error_message TEXT,
        
        -- Analytics et performance

        published_at TIMESTAMP,

    print("‚úÖ Table publication_schedule cr√©√©e")

class MYMDatabaseManager:
    """Manager principal pour toutes les op√©rations de base de donn√©es"""
    
    def __init__(self, db_path="mym_automation.db"):
        self.db_path = db_path
        self.connection = None
        self.initialize_database()
    
    def add_content(self, content_data):
        """Ajout nouveau contenu √† la biblioth√®que"""
        sql = """
        INSERT INTO content_library (
            content_id, content_type, file_path, title, description,
            tags, nsfw_level, persona_id, prompt_used, model_used,
            quality_score, status
        ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
        """
        
        try:
            self.connection.execute(sql, (
                content_data['content_id'],
                content_data['content_type'],
                content_data['file_path'],
                content_data['title'],
                content_data.get('description', ''),
                json.dumps(content_data.get('tags', [])),
                content_data.get('nsfw_level', 1),
                content_data.get('persona_id', 'leana_mystere'),
                content_data.get('prompt_used', ''),
                content_data.get('model_used', ''),
                content_data.get('quality_score', 0.0),
                content_data.get('status', 'generated')
            ))
            self.connection.commit()
            print(f"‚úÖ Contenu ajout√©: {content_data['content_id']}")
            return True
        except sqlite3.Error as e:
            print(f"‚ùå Erreur ajout contenu: {e}")
            return False
    
    def schedule_publication(self, schedule_data):
        """Planification d'une publication"""
        sql = """
        INSERT INTO publication_schedule (
            schedule_id, content_id, scheduled_datetime, platform,
            publication_type, caption, pricing_tier, status
        ) VALUES (?, ?, ?, ?, ?, ?, ?, ?)
        """
        
        try:
            self.connection.execute(sql, (
                schedule_data['schedule_id'],
                schedule_data['content_id'],
                schedule_data['scheduled_datetime'],
                schedule_data.get('platform', 'mym'),
                schedule_data.get('publication_type', 'post'),
                schedule_data.get('caption', ''),
                schedule_data.get('pricing_tier', 'premium'),
                'scheduled'
            ))
            self.connection.commit()
            print(f"‚úÖ Publication planifi√©e: {schedule_data['schedule_id']}")
            return True
        except sqlite3.Error as e:
            print(f"‚ùå Erreur planification: {e}")
            return False
    
    def get_pending_publications(self):
        """R√©cup√©ration des publications en attente"""
        sql = """
        SELECT ps.*, cl.file_path, cl.title, cl.content_type
        FROM publication_schedule ps
        JOIN content_library cl ON ps.content_id = cl.content_id
        WHERE ps.status = 'scheduled' 
        AND ps.scheduled_datetime <= datetime('now', '+1 hour')
        ORDER BY ps.scheduled_datetime ASC
        """
        
        cursor = self.connection.execute(sql)
        results = cursor.fetchall()
        
        return [dict(row) for row in results]
    
    def log_interaction(self, interaction_data):
        """Enregistrement d'une interaction abonn√©"""
        sql = """
        INSERT INTO subscriber_interactions (
            interaction_id, subscriber_id, subscriber_username,
            interaction_type, message_content, sentiment_score,
            engagement_score
        ) VALUES (?, ?, ?, ?, ?, ?, ?)
        """
        
        try:
            self.connection.execute(sql, (
                interaction_data['interaction_id'],
                interaction_data['subscriber_id'],
                interaction_data.get('subscriber_username', 'Anonymous'),
                interaction_data['interaction_type'],
                interaction_data.get('message_content', ''),
                interaction_data.get('sentiment_score', 0.0),
                interaction_data.get('engagement_score', 0.0)
            ))
            self.connection.commit()
            return True
        except sqlite3.Error as e:
            print(f"‚ùå Erreur log interaction: {e}")
            return False

def create_backup(self):
    """Cr√©ation backup automatique"""
    import shutil
    from datetime import datetime
    
    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    backup_name = f"mym_backup_{timestamp}.db"
    backup_path = os.path.join("mym_data/backups", backup_name)
    
    try:
        shutil.copy2(self.db_path, backup_path)
        print(f"‚úÖ Backup cr√©√©: {backup_path}")
        return backup_path
    except Exception as e:
        print(f"‚ùå Erreur backup: {e}")
        return None

def get_database_stats(self):
    """Statistiques de la base"""
    stats = {}
    
    # Contenu total
    cursor = self.connection.execute(
        "SELECT COUNT(*) FROM content_library"
    )
    stats['total_content'] = cursor.fetchone()[0]
    
    # Publications programm√©es
    cursor = self.connection.execute(
        "SELECT COUNT(*) FROM publication_schedule WHERE status='scheduled'"
    )
    stats['scheduled_posts'] = cursor.fetchone()[0]
    
    # Interactions aujourd'hui
    cursor = self.connection.execute(
        "SELECT COUNT(*) FROM subscriber_interactions WHERE DATE(interaction_timestamp) = DATE('now')"
    )
    stats['today_interactions'] = cursor.fetchone()[0]
    
    return stats
