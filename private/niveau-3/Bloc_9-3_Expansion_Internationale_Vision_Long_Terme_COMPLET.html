Pour les Débutants (Même 70 ans) - Qu'est-ce que l'Expansion Internationale ?

L'expansion internationale signifie adapter votre IA Léana Mystère pour différents pays. Par exemple, créer une version allemande qui parle allemand et comprend la culture allemande, ou une version américaine adaptée aux goûts américains. C'est comme avoir plusieurs "filles" de Léana, chacune parfaitement adaptée à son pays.

            base_persona, market_config

    def _adapt_personality_traits(self, base_personality: Dict, cultural_traits: List[str]) -> Dict[str, str]:
        """Adapte la personnalité selon les traits culturels"""
        
        adapted_traits = base_personality.copy()
        
        # Ajustements culturels
        cultural_adjustments = {
            'sophistication': 'sophistiquée et raffinée',

        adapted_traits['cultural_identity'] = ' • '.join(personality_description)
        
        return adapted_traits

    def _adapt_pricing_for_market(self, base_pricing: Dict, target_market: str) -> Dict[str, float]:
        """Adapte les prix selon le pouvoir d'achat du marché"""
        
        # Multiplicateurs par marché (basés sur pouvoir d'achat et concurrence)
        market_multipliers = {
            'france': 1.0,    # Base européenne
            'germany': 1.2,   # Pouvoir d'achat plus élevé
            'spain': 0.8,     # Pouvoir d'achat plus faible
            'usa': 1.5        # Marché premium, concurrence élevée
        }
        
        multiplier = market_multipliers.get(target_market, 1.0)
        
        adapted_pricing = {}
        for tier, price in base_pricing.items():
            adapted_pricing[tier] = round(price * multiplier, 2)
        
        return adapted_pricing

    def _estimate_market_revenue(self, base_persona: Dict, market_config: MarketConfig, pricing: Dict) -> float:
        """Estime les revenus potentiels sur le marché"""
        
        # Facteurs de calcul
        market_size_factor = {
            'france': 1.0,    # 67M habitants
            'germany': 1.2,   # 83M habitants  
            'spain': 0.7,     # 47M habitants
            'usa': 4.5        # 330M habitants
        }
        
        competition_penalty = (10 - market_config.competition_level) / 10
        entry_difficulty_penalty = (10 - market_config.entry_difficulty) / 10
        
        base_revenue = pricing['premium'] * 100  # Estimation 100 clients premium/mois
        
        market_factor = market_size_factor.get(market_config.country_code.lower(), 1.0)
        
        estimated_monthly = base_revenue * market_factor * competition_penalty * entry_difficulty_penalty
        
        return round(estimated_monthly, 2)

    def _get_cultural_color_palette(self, beauty_standards: Dict) -> str:
        """Retourne une palette de couleurs culturellement appropriée"""
        
        style = beauty_standards.get('style', 'naturel').lower()
        
        if 'parisienne' in style or 'sophistiqué' in style:
            return "noir, blanc, rouge, beige"
        elif 'naturel' in style or 'authentique' in style:
            return "tons neutres, beige, marron, vert"
        elif 'méditerranéenne' in style:
            return "rouge, orange, doré, terracotta"
        elif 'diversité' in style:
            return "palette variée selon persona"
        else:
            return "tons naturels harmonieux"

        filename = f"{persona.persona_id}_config.json"
        filepath = os.path.join(output_dir, filename)
        
        # Conversion en dictionnaire pour JSON
        persona_dict = asdict(persona)
        
        # Ajout métadonnées
        persona_dict['export_metadata'] = {
            'created_by': 'GlobalExpansionMasterPlan',
            'version': '1.0',
            'export_date': datetime.now().isoformat(),
            'format_version': 'v2025.1'
        }
        
        # Sauvegarde
        with open(filepath, 'w', encoding='utf-8') as f:
            json.dump(persona_dict, f, indent=2, ensure_ascii=False)
        
        print(f"💾 Persona sauvegardé: {filepath}")
        
        return filepath

    def generate_market_analysis_report(self, target_markets: List[str]) -> Dict:
        """Génère un rapport d'analyse des marchés"""
        
        print("\n📊 Génération rapport d'analyse des marchés...")
        
        report = {
            'analysis_date': datetime.now().isoformat(),
            'markets_analyzed': len(target_markets),
            'markets_details': {},
            'recommendations': {},
            'investment_summary': {}
        }
        
        total_investment = 0
        total_estimated_revenue = 0
        
        for market in target_markets:
            if market not in self.markets_config:
                continue
                
            config = self.markets_config[market]
            
            # Calcul investissement estimé selon taille marché
            market_investment = self._calculate_market_investment(config)
            total_investment += market_investment
            
            # Revenus estimés
            market_revenue = self._estimate_total_market_revenue(config)
            total_estimated_revenue += market_revenue
            
            report['markets_details'][market] = {
                'country': config.country_name,
                'market_size': config.market_size,
                'competition_level': config.competition_level,
                'entry_difficulty': config.entry_difficulty,
                'estimated_investment': market_investment,
                'estimated_monthly_revenue': market_revenue,
                'roi_projection': round(market_revenue * 12 / market_investment, 2),
                'platforms': config.platform_preferences,
                'key_requirements': config.legal_requirements
            }
            
            # Recommandations spécifiques
            report['recommendations'][market] = self._generate_market_recommendations(config)
        
        # Résumé financier
        report['investment_summary'] = {
            'total_investment_required': total_investment,
            'total_monthly_revenue_estimated': total_estimated_revenue,
            'annual_revenue_projection': total_estimated_revenue * 12,
            'global_roi_projection': round(total_estimated_revenue * 12 / total_investment, 2),
            'break_even_months': round(total_investment / total_estimated_revenue, 1)
        }
        
        print(f"✅ Rapport généré pour {len(target_markets)} marchés")
        print(f"💰 Investissement total: {total_investment:,.0f}€")
        print(f"📈 Revenus annuels estimés: {total_estimated_revenue * 12:,.0f}€")
        
        return report

    def _calculate_market_investment(self, config: MarketConfig) -> float:
        """Calcule l'investissement nécessaire pour un marché"""
        
        base_investment = 10000  # Base 10k€
        
        # Ajustements selon difficulté d'entrée
        difficulty_multiplier = config.entry_difficulty / 10
        competition_multiplier = config.competition_level / 10
        
        market_investment = base_investment * (1 + difficulty_multiplier + competition_multiplier)
        
        return round(market_investment, 2)

    def _estimate_total_market_revenue(self, config: MarketConfig) -> float:
        """Estime les revenus totaux mensuels sur un marché"""
        
        # Estimation basée sur la taille de marché et la concurrence
        market_size_value = {
            '67M habitants': 5000,   # France
            '83M habitants': 6000,   # Allemagne
            '47M habitants': 3500,   # Espagne  
            '330M habitants': 15000  # USA
        }
        
        base_revenue = market_size_value.get(config.market_size, 4000)
        
        # Ajustement selon concurrence
        competition_factor = (10 - config.competition_level) / 10
        
        estimated_revenue = base_revenue * competition_factor
        
        return round(estimated_revenue, 2)

    def _generate_market_recommendations(self, config: MarketConfig) -> List[str]:
        """Génère des recommandations spécifiques par marché"""
        
        recommendations = []
        
        if config.competition_level > 8:
            recommendations.append("Forte concurrence - Nécessite différenciation forte")
            recommendations.append("Budget marketing élevé recommandé")
        
        if config.entry_difficulty > 7:
            recommendations.append("Marché complexe - Partenariat local recommandé")
            recommendations.append("Phase pilote recommandée avant déploiement complet")
        
        # Recommandations spécifiques par pays
        if config.country_code == 'DE':
            recommendations.extend([
                "Transparence totale sur nature IA obligatoire",
                "Qualité technique prime sur quantité",
                "Respect strict RGPD essentiel"
            ])
        elif config.country_code == 'US':
            recommendations.extend([
                "Stratégie multi-plateformes nécessaire", 
                "Innovation constante pour rester compétitif",
                "Veille réglementaire par État"
            ])
        elif config.country_code == 'ES':
            recommendations.extend([
                "Approche communautaire et chaleureuse",
                "Contenu expressif et passionné",
                "Timing posts selon culture méditerranéenne"
            ])
        
        return recommendations
